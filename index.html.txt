/* ---------- Astronomical / calendar functions (same as before) ---------- */
function jdFromDate(dd, mm, yy) {
  var a = Math.floor((14 - mm) / 12);
  var y = yy + 4800 - a;
  var m = mm + 12 * a - 3;
  var jd = dd + Math.floor((153 * m + 2) / 5) + 365 * y + Math.floor(y / 4) - Math.floor(y / 100) + Math.floor(y / 400) - 32045;
  if (jd < 2299161) {
    jd = dd + Math.floor((153 * m + 2) / 5) + 365 * y + Math.floor(y / 4) - 32083;
  }
  return jd;
}
function getNewMoonDay(k, timeZone) {
  var T = k / 1236.85;
  var T2 = T * T;
  var T3 = T2 * T;
  var dr = Math.PI / 180;
  var Jd1 = 2415020.75933 + 29.53058868 * k + 0.0001178 * T2 - 0.000000155 * T3;
  Jd1 += 0.00033 * Math.sin((166.56 + 132.87 * T - 0.009173 * T2) * dr);
  var M = 359.2242 + 29.10535608 * k - 0.0000333 * T2 - 0.00000347 * T3;
  var Mpr = 306.0253 + 385.81691806 * k + 0.0107306 * T2 + 0.00001236 * T3;
  var F = 21.2964 + 390.67050646 * k - 0.0016528 * T2 - 0.00000239 * T3;
  var C1 = (0.1734 - 0.000393 * T) * Math.sin(M * dr) + 0.0021 * Math.sin(2 * dr * M);
  C1 -= 0.4068 * Math.sin(Mpr * dr) + 0.0161 * Math.sin(dr * 2 * Mpr);
  C1 -= 0.0004 * Math.sin(dr * 3 * Mpr);
  C1 += 0.0104 * Math.sin(dr * 2 * F) - 0.0051 * Math.sin(dr * (M + Mpr));
  C1 -= 0.0074 * Math.sin(dr * (M - Mpr)) + 0.0004 * Math.sin(dr * (2 * F + M));
  C1 -= 0.0004 * Math.sin(dr * (2 * F - M)) - 0.0006 * Math.sin(dr * (2 * F + Mpr));
  C1 += 0.0010 * Math.sin(dr * (2 * F - Mpr)) + 0.0005 * Math.sin(dr * (2 * Mpr + M));
  var deltat;
  if (T < -11) {
    deltat = 0.001 + 0.000839 * T + 0.0002261 * T2 - 0.00000845 * T3 - 0.000000081 * T * T3;
  } else {
    deltat = -0.000278 + 0.000265 * T + 0.000262 * T2;
  }
  var JdNew = Jd1 + C1 - deltat;
  return Math.floor(JdNew + 0.5 + timeZone / 24);
}
function getSunLongitude(jdn, timeZone) {
  var T = (jdn - 2451545.5 - timeZone / 24) / 36525;
  var T2 = T * T;
  var dr = Math.PI / 180;
  var M = 357.52910 + 35999.05030 * T - 0.0001559 * T2 - 0.00000048 * T * T2;
  var L0 = 280.46645 + 36000.76983 * T + 0.0003032 * T2;
  var DL = (1.914600 - 0.004817 * T - 0.000014 * T2) * Math.sin(dr * M);
  DL += (0.019993 - 0.000101 * T) * Math.sin(dr * 2 * M) + 0.000290 * Math.sin(dr * 3 * M);
  var L = L0 + DL;
  L = L * dr;
  L = L - Math.PI * 2 * (Math.floor(L / (Math.PI * 2)));
  return Math.floor(L / Math.PI * 6);
}
function getLunarMonth11(yy, timeZone) {
  var off = jdFromDate(31, 12, yy) - 2415021;
  var k = Math.floor(off / 29.530588853);
  var nm = getNewMoonDay(k, timeZone);
  var sunLong = getSunLongitude(nm, timeZone); // sun longitude at new moon
  if (sunLong >= 9) {
    nm = getNewMoonDay(k - 1, timeZone);
  }
  return nm;
}
function getLeapMonthOffset(a11, timeZone) {
  var k = Math.floor((a11 - 2415021.076998695) / 29.530588853 + 0.5);
  var last = 0;
  var i = 1;
  var arc = getSunLongitude(getNewMoonDay(k + i, timeZone), timeZone);
  do {
    last = arc;
    i++;
    arc = getSunLongitude(getNewMoonDay(k + i, timeZone), timeZone);
  } while (arc !== last && i < 14);
  return i - 1;
}
function convertSolar2Lunar(dd, mm, yy, timeZone) {
  var dayNumber = jdFromDate(dd, mm, yy);
  var k = Math.floor((dayNumber - 2415021.076998695) / 29.530588853);
  var monthStart = getNewMoonDay(k + 1, timeZone);
  if (monthStart > dayNumber) {
    monthStart = getNewMoonDay(k, timeZone);
  }
  var a11 = getLunarMonth11(yy, timeZone);
  var b11 = a11;
  var lunarYear;
  if (a11 >= monthStart) {
    lunarYear = yy;
    a11 = getLunarMonth11(yy - 1, timeZone);
  } else {
    lunarYear = yy + 1;
    b11 = getLunarMonth11(yy + 1, timeZone);
  }
  var lunarDay = dayNumber - monthStart + 1;
  var diff = Math.floor((monthStart - a11) / 29);
  var lunarLeap = 0;
  var lunarMonth = diff + 11;
  if (b11 - a11 > 365) {
    var leapMonthDiff = getLeapMonthOffset(a11, timeZone);
    if (diff >= leapMonthDiff) {
      lunarMonth = diff + 10;
      if (diff === leapMonthDiff) lunarLeap = 1;
    }
  }
  if (lunarMonth > 12) lunarMonth = lunarMonth - 12;
  if (lunarMonth >= 11 && diff < 4) lunarYear -= 1;
  return { day: lunarDay, month: lunarMonth, year: lunarYear, leap: lunarLeap };
}

/* ---------- Utility: get CanChi & index from lunar year ---------- */
function getCanChiAndIndex(lunarYear){
  // base reference: 1984 = Giáp Tý index 0
  const base = 1984;
  let idx = (lunarYear - base) % 60;
  if (idx < 0) idx += 60;
  // Thiên Can and Địa Chi calculation (standard)
  const can = canNames[(lunarYear + 6) % 10];
  const chi = chiNames[(lunarYear + 8) % 12];
  return { canChi: can + " " + chi, index: idx };
}

/* ---------- Rendering functions ---------- */
function renderInterpretationByIndex(i){
  const data = interpretations60[i];
  if(!data) return;
  document.getElementById("napAm").textContent = data.title || napAmNames[i] || ("Nạp Âm " + (i+1));
  for(let k=1;k<=9;k++){
    const el = document.getElementById("s"+k);
    el.textContent = data["s"+k] || "";
  }
}

function renderFullList(){
  const container = document.getElementById("fullInterpretations");
  container.innerHTML = '';
  interpretations60.forEach((it, idx) => {
    const sec = document.createElement('section');
    sec.style.padding = "12px";
    sec.style.borderBottom = "1px solid #eee";
    const h = document.createElement('h3');
    h.textContent = (idx+1) + ". " + (it.title || ("Nạp Âm " + (idx+1)));
    h.style.margin = "8px 0";
    sec.appendChild(h);
    for(let k=1;k<=9;k++){
      const subh = document.createElement('h4');
      subh.textContent = k + ". " + (["Tính cách","Sự nghiệp","Tài lộc","Gia đạo","Tình duyên","Sức khỏe","Màu hợp – màu kỵ","Hướng hợp – hướng xung","Mệnh tương hợp – tương khắc"][k-1]);
      subh.style.margin = "6px 0 0 0";
      subh.style.fontSize = "14px";
      const p = document.createElement('p');
      p.textContent = it["s"+k] || "";
      p.style.margin = "4px 0 8px 0";
      sec.appendChild(subh);
      sec.appendChild(p);
    }
    container.appendChild(sec);
  });
}

/* ---------- Event binding ---------- */
document.getElementById("btnCalc").addEventListener("click", function(){
  const d = document.getElementById("solarDate").value;
  if (!d) {
    alert("Vui lòng chọn ngày dương lịch (ví dụ: 1987-01-19).");
    return;
  }
  const parts = d.split("-");
  const yy = parseInt(parts[0],10);
  const mm = parseInt(parts[1],10);
  const dd = parseInt(parts[2],10);
  const timeZone = 7.0; // GMT+7
  const lunar = convertSolar2Lunar(dd, mm, yy, timeZone);
  const lunarText = lunar.day + "/" + lunar.month + "/" + lunar.year + (lunar.leap ? " (nhuận)" : "");
  document.getElementById("lunarDate").textContent = lunarText;

  const cc = getCanChiAndIndex(lunar.year);
  document.getElementById("canChi").textContent = cc.canChi + " (năm âm: " + lunar.year + ")";
  document.getElementById("jiaziIndex").textContent = cc.index + " (0→Giáp Tý)";

  // Determine nap am from index
  const index = cc.index;
  const napamName = napAmNames[index] || ("Nạp Âm " + (index+1));
  document.getElementById("napAm").textContent = napamName;

  // render interpretation for this index
  renderInterpretationByIndex(index);

  // scroll result into view
  document.getElementById("result").scrollIntoView({behavior:"smooth"});
});
